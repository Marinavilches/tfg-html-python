from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, roc_auc_score
import pandas as pd
import numpy as np

# Función para evaluar modelos de clasificación binaria
def evaluar_clasificacion(y_real, y_pred, y_proba, nombre_modelo):
    accuracy = accuracy_score(y_real, y_pred)
    precision = precision_score(y_real, y_pred)
    recall = recall_score(y_real, y_pred)
    f1 = f1_score(y_real, y_pred)
    
    if y_proba is not None:
        roc_auc = roc_auc_score(y_real, y_proba)
    else:
        roc_auc = np.nan
    
    return {
        "Modelo": nombre_modelo,
        "Accuracy": accuracy,
        "Precision": precision,
        "Recall": recall,
        "F1-score": f1,
        "ROC-AUC": roc_auc
    }

from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import OneHotEncoder
from sklearn.tree import DecisionTreeClassifier

# Preprocesado para variables numéricas
numeric_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="median"))
])

# Preprocesado para variables categóricas
categorical_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot", OneHotEncoder(handle_unknown="ignore"))
])

# ColumnTransformer
preprocessor = ColumnTransformer(
    transformers=[
        ("num", numeric_transformer, numeric_features),
        ("cat", categorical_transformer, categorical_features)
    ]
)

# Pipeline: preprocesado + árbol clasificador
tree_classifier = Pipeline(steps=[
    ("preprocessor", preprocessor),
    ("model", DecisionTreeClassifier(random_state=42))
])

# Entrenar modelo
tree_classifier.fit(X_train, y_train)

# Predicciones
y_pred_tree_train = tree_classifier.predict(X_train)
y_pred_tree_test = tree_classifier.predict(X_test)

# Nombres tras el preprocesado
feature_names = preprocessor_rf_clf.get_feature_names_out()
importances = model_rf_clf.feature_importances_

df_importancias_clf = pd.DataFrame({
    "Variable_transformada": feature_names,
    "Importancia": importances
})

# Limpiar nombres agrupando variables categóricas
def limpiar_nombre_variable(nombre):
    if nombre.startswith("num__"):
        return nombre.replace("num__", "")
    
    if nombre.startswith("cat__"):
        nombre_limpio = nombre.replace("cat__", "")
        
        for cat in categorical_features:
            if nombre_limpio.startswith(cat + "_"):
                return cat
        
        return nombre_limpio
    
    return nombre

df_importancias_clf["Variable"] = df_importancias_clf["Variable_transformada"].apply(limpiar_nombre_variable)

# Agrupar importancias por variable original
df_importancias_clf_agrupadas = (
    df_importancias_clf
    .groupby("Variable", as_index=False)["Importancia"]
    .sum()

from sklearn.metrics import mean_absolute_error, mean_squared_error
import pandas as pd
import numpy as np

resultados_gps_regresion = []

gps_disponibles = (
    F1_modelizacion[F1_modelizacion["Year"] >= 2023][["Year", "Round"]]
    .drop_duplicates()
    .sort_values(["Year", "Round"])
)

for _, row in gps_disponibles.iterrows():
    year_gp = row["Year"]
    round_gp = row["Round"]
    
    gp_temp = F1_modelizacion[
        (F1_modelizacion["Year"] == year_gp) &
        (F1_modelizacion["Round"] == round_gp)
    ].copy()
    
    # Crear X sin variables objetivo
    columnas_excluir = ["PositionOrder", "PointsZone"]
    X_gp_temp = gp_temp.drop(columns=columnas_excluir)
    
    # Predicción de posición
    gp_temp["PositionOrder_pred"] = modelo_regresion.predict(X_gp_temp)
    
    # Redondear predicción a posición entera
    gp_temp["PositionOrder_pred_round"] = (
        gp_temp["PositionOrder_pred"]
        .round()
        .clip(lower=1, upper=22)
        .astype(int)
    )
    
    # Errores
    gp_temp["Error_abs"] = abs(gp_temp["PositionOrder"] - gp_temp["PositionOrder_pred"])
    gp_temp["Error_abs_round"] = abs(gp_temp["PositionOrder"] - gp_temp["PositionOrder_pred_round"])
    
    mae_gp = mean_absolute_error(gp_temp["PositionOrder"], gp_temp["PositionOrder_pred"])
    rmse_gp = np.sqrt(mean_squared_error(gp_temp["PositionOrder"], gp_temp["PositionOrder_pred"]))
    
    coincidencias_exactas = (gp_temp["Error_abs_round"] == 0).sum()
    coincidencias_margen_1 = (gp_temp["Error_abs_round"] <= 1).sum()
    coincidencias_margen_2 = (gp_temp["Error_abs_round"] <= 2).sum()
    
    resultados_gps_regresion.append({
        "Year": year_gp,
        "Round": round_gp,
        "N_pilotos": len(gp_temp),
        "MAE": mae_gp,
        "RMSE": rmse_gp,
        "Coincidencias_exactas": coincidencias_exactas,
        "Coincidencias_±1": coincidencias_margen_1,
        "Coincidencias_±2": coincidencias_margen_2
    })

comparacion_gps_regresion = pd.DataFrame(resultados_gps_regresion)

# Ordenar por mejor ajuste:
# 1º más coincidencias ±2
# 2º más coincidencias ±1
# 3º más exactas
# 4º menor MAE
comparacion_gps_regresion_ordenado = comparacion_gps_regresion.sort_values(
    by=["Coincidencias_±2", "Coincidencias_±1", "Coincidencias_exactas", "MAE"],
    ascending=[False, False, False, True]
)
