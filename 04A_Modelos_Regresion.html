from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
import numpy as np
import pandas as pd

# Función para evaluar modelos de regresión
def evaluar_regresion(y_real, y_pred, nombre_modelo):
    mae = mean_absolute_error(y_real, y_pred)
    mse = mean_squared_error(y_real, y_pred)
    rmse = np.sqrt(mse)
    r2 = r2_score(y_real, y_pred)
    
    # Métricas relativas respecto al modelo que predice la media
    rae = np.sum(np.abs(y_real - y_pred)) / np.sum(np.abs(y_real - np.mean(y_real)))
    rrse = np.sqrt(np.sum((y_real - y_pred) ** 2)) / np.sqrt(np.sum((y_real - np.mean(y_real)) ** 2))
    
    return {
        "Modelo": nombre_modelo,
        "MAE": mae,
        "RMSE": rmse,
        "R2": r2,
        "RAE": rae,
        "RRSE": rrse
    }

# Modelo base: predecir siempre la media del conjunto de entrenamiento
pred_media = np.full(shape=len(y_test), fill_value=y_train.mean())

resultado_base = evaluar_regresion(
    y_real=y_test,
    y_pred=pred_media,
    nombre_modelo="Modelo base media"
)

resultados_regresion = pd.DataFrame([resultado_base])

display(resultados_regresion.round(4))

from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import OneHotEncoder
from sklearn.tree import DecisionTreeRegressor

# Preprocesado para variables numéricas
numeric_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="median"))
])

# Preprocesado para variables categóricas
categorical_transformer = Pipeline(steps=[
    ("imputer", SimpleImputer(strategy="most_frequent")),
    ("onehot", OneHotEncoder(handle_unknown="ignore"))
])

# ColumnTransformer: aplica cada tratamiento según el tipo de variable
preprocessor = ColumnTransformer(
    transformers=[
        ("num", numeric_transformer, numeric_features),
        ("cat", categorical_transformer, categorical_features)
    ]
)

# Pipeline completo: preprocesado + modelo
tree_regressor = Pipeline(steps=[
    ("preprocessor", preprocessor),
    ("model", DecisionTreeRegressor(random_state=42))
])

# Entrenar modelo
tree_regressor.fit(X_train, y_train)

# Predicciones
y_pred_tree_train = tree_regressor.predict(X_train)
y_pred_tree_test = tree_regressor.predict(X_test)

    y_real=y_train,

)

)

import matplotlib.pyplot as plt
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error

depth_values = [2, 3, 4, 5, 6, 8, 10, 12, 15, None]

mae_train_rf_overfit = []
mae_test_rf_overfit = []

for depth in depth_values:
    model = Pipeline(steps=[
        ("preprocessor", preprocessor),
        ("model", RandomForestRegressor(
            n_estimators=200,
            max_depth=depth,
            min_samples_leaf=1,
            min_samples_split=2,
            max_features=1.0,
            random_state=42,
            n_jobs=-1
        ))
    ])
    
    model.fit(X_train, y_train)
    
    pred_train = model.predict(X_train)
    pred_test = model.predict(X_test)
    
    mae_train_rf_overfit.append(mean_absolute_error(y_train, pred_train))
    mae_test_rf_overfit.append(mean_absolute_error(y_test, pred_test))

depth_labels = [str(d) if d is not None else "None" for d in depth_values]

plt.figure(figsize=(8, 5))
plt.plot(depth_labels, mae_train_rf_overfit, marker="o", label="Train")
plt.plot(depth_labels, mae_test_rf_overfit, marker="o", label="Test")

plt.xlabel("Profundidad máxima")
plt.ylabel("MAE")
plt.title("Evolución del MAE según la profundidad - Random Forest sin regularizar")
plt.grid(alpha=0.3)
plt.legend()
plt.tight_layout()

plt.savefig("mae_profundidad_rf_sin_regularizar.png", dpi=300, bbox_inches="tight")
plt.show()
